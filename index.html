Functions as Parameters


					function add(x: number, y: number): number {
						return x + y;
					}

					function calculate(x: number, y: number, operation: (a: number, b: number) => number): number {
						return operation(x, y);
					}

					console.log(calculate(1, 2, add)); // Output: 3

Optional Parameters in Functions


					function greet(name: string, greeting?: string): string {
						if (greeting) {
							return `${greeting}, ${name}!`;
						} else {
							return `Hello, ${name}!`;
						}
					}

					console.log(greet("John")); // Output: Hello, John!
					console.log(greet("John", "Hi")); // Output: Hi, John!

Void Type

Never Type (Function Return)

					let value: unknown = "Hello World";

Rest and Spread


					function sum(...numbers: number[]) {
						return numbers.reduce((total, num) => total + num);
					}

					console.log(sum(1, 2, 3, 4)); // Output: 10
				

					const numbers = [1, 2, 3, 4];
					const newNumbers = [...numbers, 5, 6];

					console.log(newNumbers); // Output: [1, 2, 3, 4, 5, 6]

Function Overloads


					function add(a: number, b: number): number;
					function add(a: string, b: string): string;
					function add(a: any, b: any): any {
						return a + b;
					}

					console.log(add(1, 2)); // Output: 3
					console.log(add('Hello', 'World')); // Output: HelloWorld

Narrowing


					let x: string | number;

					if (typeof x === 'string') {
						console.log(x.toUpperCase()); // Output: ERROR - Property 'toUpperCase' does not exist on type 'number'.
					} else {
						console.log(x.toFixed(2)); // Output: ERROR - Property 'toFixed' does not exist on type 'string'.
					}

Access Modifiers


					class Person {
						public name: string;
						private age: number;
						constructor(name: string, age: number) {
							this.name = name;
							this.age = age;
						}
						greet() {
							console.log(`Hello, my name is ${this.name} and I am ${this.age} years old.`);
						}
					}

Inheritance


					class Animal {
						name: string;

						constructor(name: string) {
							this.name = name;
						}

						move(distance: number) {
							console.log(`${this.name} moved ${distance} meters.`);
						}
					}

					class Dog extends Animal {
						breed: string;

						constructor(name: string, breed: string) {
							super(name);
							this.breed = breed;
						}

						bark() {
							console.log('Woof! Woof!');
						}
					}

					let dog = new Dog('Buddy', 'Golden Retriever');
					dog.move(10); // Output: Buddy moved 10 meters.
					dog.bark(); // Output: Woof! Woof!

Static Methods and Properties


					class Calculator {
						static PI: number = 3.14;

						static add(x: number, y: number) {
							return x + y;
						}
					}

					console.log(Calculator.PI); // Output: 3.14
					console.log(Calculator.add(2, 3)); // Output: 5

Abstract Classes


					abstract class Shape {
						abstract getArea(): number;
					}

					class Rectangle extends Shape {
						width: number;
						height: number;

						constructor(width: number, height: number) {
							super();
							this.width = width;
							this.height = height;
						}

						getArea() {
							return this.width * this.height;
						}
					}

					let rect = new Rectangle(5, 10);
					console.log(rect.getArea()); // Output: 50

Introduction to Generics


						function identity(arg: T): T {
							return arg;
						}

						console.log(identity('Hello')); // Output: Hello
						console.log(identity(42)); // Output: 42

Type Generics


						function add(x: T, y: T): T {
							return x + y;
						}

						console.log(add(2, 3)); // Output: 5
						console.log(add('Hello', 'World')); // Output: HelloWorld

Function Generics


						function map(arr: T[], fn: (item: T) => U): U[] {
							return arr.map(fn);
						}

						let nums = [1, 2, 3];
						let doubled = map(nums, (num) => num * 2);
						console.log(doubled); // Output: [2, 4, 6]

Multiple Parameter Generics


						function zip(arr1: T[], arr2: U[]): [T, U][] {
							return arr1.map((item, i) => [item, arr2[i]]);
						}

						let names = ['John', 'Jane', 'Bob'];
						let ages = [30, 25, 40];
						let people = zip(names, ages);
						console.log(people); // Output: [["John", 30], ["Jane", 25], ["Bob", 40]]

Asynchronous Functions


					function delay(ms: number): Promise {
						return new Promise((resolve) => setTimeout(resolve, ms));
					}

					async function run() {
						console.log('Starting...');
						await delay(2000);
						console.log('Finished.');
					}

					run();
